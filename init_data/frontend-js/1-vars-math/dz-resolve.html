<script>

    var pocket=prompt('input amount of money in your pocket')

    var choc=prompt('input price of the chocolate')

    var buyAble=(pocket/choc)-(pocket/choc)%1

    var left =pocket-buyAble*choc

    alert('left in pocket:'+left)
